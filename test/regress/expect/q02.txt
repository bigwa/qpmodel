select
	s_acctbal,
	s_name,
	n_name,
	p_partkey,
	p_mfgr,
	s_address,
	s_phone,
	s_comment
from
	part,
	supplier,
	partsupp,
	nation,
	region
where
	p_partkey = ps_partkey
	and s_suppkey = ps_suppkey
	and p_size = 15
	and p_type like '%BRASS'
	and s_nationkey = n_nationkey
	and n_regionkey = r_regionkey
	and r_name = 'EUROPE'
	and ps_supplycost = (
		select
			min(ps_supplycost)
		from
			partsupp,
			supplier,
			nation,
			region
		where
			p_partkey = ps_partkey
			and s_suppkey = ps_suppkey
			and s_nationkey = n_nationkey
			and n_regionkey = r_regionkey
			and r_name = 'EUROPE'
	)
order by
	s_acctbal desc,
	n_name,
	s_name,
	p_partkey
limit 100

    Output: supplier.s_acctbal[0],supplier.s_name[1],nation.n_name[2],part.p_partkey[3],part.p_mfgr[4],supplier.s_address[5],supplier.s_phone[6],supplier.s_comment[7]

        Order by: supplier.s_acctbal[0], nation.n_name[2], supplier.s_name[1], part.p_partkey[3]

                Filter: part.p_size[5]=15 and part.p_type[4]like'%BRASS'

                Output: supplier.s_acctbal[1],supplier.s_name[2],nation.n_name[3],supplier.s_address[4],supplier.s_phone[5],supplier.s_comment[6],partsupp.ps_partkey[7]
                Filter: nation.n_regionkey[8]=region.r_regionkey[0]

                    Output: region.r_regionkey[0]
                    Filter: region.r_name[1]='EUROPE'
                -> PhysicHashJoin  (actual rows=0)
                    Output: supplier.s_acctbal[3],supplier.s_name[4],nation.n_name[0],supplier.s_address[5],supplier.s_phone[6],supplier.s_comment[7],partsupp.ps_partkey[8],nation.n_regionkey[1]
                    Filter: supplier.s_nationkey[9]=nation.n_nationkey[2]
                    -> PhysicScanTable nation (actual rows=0)
                        Output: nation.n_name[1],nation.n_regionkey[2],nation.n_nationkey[0]
                    -> PhysicHashJoin  (actual rows=0)
                        Output: supplier.s_acctbal[2],supplier.s_name[3],supplier.s_address[4],supplier.s_phone[5],supplier.s_comment[6],partsupp.ps_partkey[0],supplier.s_nationkey[7]
                        Filter: supplier.s_suppkey[8]=partsupp.ps_suppkey[1]

                            Output: partsupp.ps_partkey[0],partsupp.ps_suppkey[1]

                            <ScalarSubqueryExpr> 1
                                -> PhysicHashAgg  (actual rows=0)

                                            Filter: ?part.p_partkey[0]=partsupp__1.ps_partkey[0]
                                        -> PhysicHashJoin  (actual rows=0)
                                            Output: supplier__1.s_suppkey[1]
                                            Filter: nation__1.n_regionkey[2]=region__1.r_regionkey[0]
